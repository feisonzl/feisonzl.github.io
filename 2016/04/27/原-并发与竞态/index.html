### 3 自旋锁

<pre class="prettyprint">`自旋锁在获取锁之前一直进入忙循环并重复检查锁是否被解锁，可用于不能休眠的流程中，例如中断处理等。
拥有自旋锁时禁止中断；
自旋锁拥有时间尽可能短；
自旋锁下的执行函数不能休眠；
<span class="hljs-number">1.</span>普通自旋锁
<span class="hljs-preprocessor">#include &lt;linux/spinlock.h&gt;</span>
<span class="hljs-preprocessor">#<span class="hljs-keyword">if</span> LINUX_VERSION_CODE &lt; KERNEL_VERSION(2,6,19)</span>
spinlock_t  <span class="hljs-keyword">lock</span>=SPIN_LOCK_UNLOCK;
<span class="hljs-preprocessor">#<span class="hljs-keyword">else</span></span>
DEFINE_SPINLOCK(<span class="hljs-keyword">lock</span>);
<span class="hljs-preprocessor">#<span class="hljs-keyword">endif</span></span>

spin_lock_init(&amp;<span class="hljs-keyword">lock</span>);

<span class="hljs-keyword">void</span> spin_lock(&amp;<span class="hljs-keyword">lock</span>);
<span class="hljs-keyword">void</span>  spin_lock_bh(&amp;<span class="hljs-keyword">lock</span>);<span class="hljs-comment">//获取自旋锁之前禁止软件中断，但硬件中断保持打开</span>
<span class="hljs-keyword">void</span> spin_lock_irq(&amp;<span class="hljs-keyword">lock</span>);<span class="hljs-comment">//获取自旋锁之前禁止中断</span>
spin_lock_irqsave(&amp;<span class="hljs-keyword">lock</span>, flags);<span class="hljs-comment">//获取自旋锁之前禁止中断，并将之前的中断状态保存在flags中</span>

<span class="hljs-keyword">void</span> spin_unlock(&amp;<span class="hljs-keyword">lock</span>);
<span class="hljs-keyword">void</span> spin_unlock_bh(&amp;<span class="hljs-keyword">lock</span>);<span class="hljs-comment">//释放自旋锁，并恢复软件中断</span>

成功返回非零值，否则返回零

<span class="hljs-preprocessor">#<span class="hljs-keyword">else</span></span>

read_lock_irq(&amp;<span class="hljs-keyword">lock</span>);<span class="hljs-comment">//获取自旋锁之前禁止中断</span>
read_lock_irqsave(&amp;<span class="hljs-keyword">lock</span>, flags);<span class="hljs-comment">//获取自旋锁之前禁止中断，并将之前的中断状态保存在flags中</span>
